# Filesystem-based bundle size measurements
python3 - <<'PY'
from pathlib import Path

for label, folder in [
    ('before', Path('/tmp/shipshape-cat2-before/web/dist/assets')),
    ('after', Path('dist/assets')),
]:
    assets = sorted(folder.glob('*.js'))
    total = sum(p.stat().st_size for p in assets)
    largest = max(assets, key=lambda p: p.stat().st_size)
    print(label, {
        'total_kb_decimal': round(total / 1000, 2),
        'chunk_count': len(assets),
        'largest_chunk': largest.name,
        'largest_chunk_kb_decimal': round(largest.stat().st_size / 1000, 2),
    })
PY

# Initial startup-path measurements from the optimized build
python3 - <<'PY'
import re
from pathlib import Path

assets = Path('dist/assets')
sizes = {p.name: p.stat().st_size for p in assets.glob('*.js')}
imports = {}
pattern = re.compile(r'''from["']\./([^"']+\.js)["']''')

for asset in assets.glob('*.js'):
    imports[asset.name] = set(pattern.findall(asset.read_text(errors='ignore')))

def closure(starts):
    seen = set()
    stack = list(starts)
    while stack:
        name = stack.pop()
        if name in seen:
            continue
        seen.add(name)
        stack.extend(imports.get(name, ()))
    return seen

for label, starts in {
    'my_week_initial': ['index-k5rnZOZq.js', 'App--Qu75XNX.js', 'MyWeekPage-CL8r3fJo.js'],
    'login_initial': ['index-k5rnZOZq.js', 'Login-B0rUCv0u.js'],
}.items():
    files = closure(starts)
    total = sum(sizes[name] for name in files)
    print(label, {
        'files': sorted(files),
        'total_kb_decimal': round(total / 1000, 2),
    })
PY
